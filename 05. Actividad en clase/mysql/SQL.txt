CREATE DATABASE PURO_GANADO;
USE PURO_GANADO;
CREATE TABLE fincas (
    id_finca INT PRIMARY KEY AUTO_INCREMENT,
    nombre_finca VARCHAR(100) NOT NULL,
    ubicacion VARCHAR(200) NOT NULL,
    area_hectareas DECIMAL(10, 2) NOT NULL,
    administrador VARCHAR(100) NOT NULL);

CREATE TABLE animales (
    id_animal INT PRIMARY KEY AUTO_INCREMENT,
    id_finca INT NOT NULL,
    codigo_chip VARCHAR(50) UNIQUE NOT NULL,
    genero VARCHAR(10) NOT NULL,
    raza VARCHAR(50) NOT NULL,
    fecha_nacimiento DATE NOT NULL,
    peso_actual DECIMAL(8, 2) NOT NULL,
    estado VARCHAR(20) DEFAULT 'activo',
    FOREIGN KEY (id_finca) REFERENCES fincas(id_finca)
);

CREATE TABLE historial_medico (
    id_historial INT PRIMARY KEY AUTO_INCREMENT,
    id_animal INT NOT NULL,
    tipo_evento VARCHAR(20) NOT NULL,
    fecha_evento DATETIME NOT NULL,
    descripcion TEXT NOT NULL,
    medicamento VARCHAR(100),
    dosis VARCHAR(50),
    via_administracion VARCHAR(20),
    veterinario VARCHAR(100) NOT NULL,
    proximo_control DATE,
    
    FOREIGN KEY (id_animal) REFERENCES animales(id_animal)
);


CREATE TABLE arbol_genealogico (
    id_relacion INT PRIMARY KEY AUTO_INCREMENT,
    id_animal INT NOT NULL UNIQUE,
    id_padre INT,
    id_madre INT,
    nivel_generacion VARCHAR(10) DEFAULT 'G1',
    registro_libro VARCHAR(50),
    
    FOREIGN KEY (id_animal) REFERENCES animales(id_animal),
    FOREIGN KEY (id_padre) REFERENCES animales(id_animal),
    FOREIGN KEY (id_madre) REFERENCES animales(id_animal)
);
